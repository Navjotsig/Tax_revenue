import itertools
import pandas as pd
import numpy as np
import statsmodels.api as sm
from statsmodels.tsa.statespace.sarimax import SARIMAX

# Define the p, d, q and P, D, Q ranges for SARIMA
p = d = q = range(0, 3)  # Adjust range if needed
P = D = Q = range(0, 2)  # Seasonal parameters
s = 12  # Set to 12 for annual seasonality; change based on your data frequency

# Generate all possible combinations of p, d, q and seasonal P, D, Q
pdq = list(itertools.product(p, d, q))
seasonal_pdq = [(x[0], x[1], x[2], s) for x in itertools.product(P, D, Q)]

# Placeholder to store results
results_list = []

# Loop through each combination of parameters
for param in pdq:
    for param_seasonal in seasonal_pdq:
        try:
            # Fit SARIMA model
            model = SARIMAX(data['Total_Tax_Revenue'],
                            order=param,
                            seasonal_order=param_seasonal,
                            enforce_stationarity=False,
                            enforce_invertibility=False)
            results = model.fit(disp=False)
            
            # Append AIC and parameters to results list
            results_list.append((results.aic, param, param_seasonal))
            print(f"SARIMA{param}x{param_seasonal}12 - AIC:{results.aic}")
        
        except Exception as e:
            print(f"Error with SARIMA{param}x{param_seasonal}12: {e}")
            continue

# Find the combination with the lowest AIC
best_model = sorted(results_list, key=lambda x: x[0])[0]
print(f"\nBest SARIMA model: ARIMA{best_model[1]}x{best_model[2]} - AIC:{best_model[0]}")
